── SEND PATTERNS ────────────────────────────────────────────────────────────────

TEXT (plain)
  sock.sendMessage(jid, { text: 'Hello' })

TEXT WITH LINK PREVIEW
  sock.sendMessage(jid, { text: 'https://...', richPreview: true })

IMAGE
  sock.sendMessage(jid, { image: Buffer|url, caption: 'text' })

IMAGE (HD)
  sock.sendMessage(jid, { image: Buffer, jpegThumbnail: hdBuffer })
  → generateProfilePictureHD for thumbnails

VIDEO
  sock.sendMessage(jid, { video: Buffer|url, caption: 'text', gifPlayback: false })

GIF (auto-play loop)
  sock.sendMessage(jid, { video: Buffer|url, gifPlayback: true })

AUDIO (voice note / PTT)
  sock.sendMessage(jid, { audio: Buffer|url, mimetype: 'audio/ogg; codecs=opus', ptt: true })

AUDIO (music / non-PTT)
  sock.sendMessage(jid, { audio: Buffer|url, mimetype: 'audio/mp4' })

DOCUMENT
  sock.sendMessage(jid, { document: Buffer|url, mimetype: '...', fileName: 'file.pdf' })

STICKER (single)
  sock.sendMessage(jid, { sticker: Buffer })

STICKER PACK (branded)
  sock.createStickerPackFromImages(images[], { packName, packAuthor })
  → returns a Buffer to send as document

ALBUM (2–10 photos/videos native)
  sock.sendMessage(jid, {
    album: [
      { image: Buffer, caption: 'optional' },
      { video: Buffer },
      ...
    ]
  })

CONTACT CARD
  sock.sendMessage(jid, {
    contacts: { displayName: 'Name', contacts: [{ vcard: 'BEGIN:VCARD...' }] }
  })

CONTACT ARRAY
  sock.sendMessage(jid, {
    contactsArray: { displayName: 'Contacts', contacts: [{ vcard }, { vcard }] }
  })

LOCATION (static)
  sock.sendMessage(jid, {
    location: { degreesLatitude: 37.7749, degreesLongitude: -122.4194,
                name: 'San Francisco', address: '...' }
  })

LIVE LOCATION
  sock.sendMessage(jid, {
    location: { degreesLatitude: lat, degreesLongitude: lng,
                isLive: true, accuracyInMeters: 10 }
  })
  → Update: sock.sendMessage(jid, { location: {...}, edit: liveLocKey })

REACTION
  sock.sendMessage(jid, { react: { text: '👍', key: msgKey } })

BULK REACTIONS (real senders)
  sock.sendBulkReactions(jid, messageId, '👍', count, false)
  ⚠ fake=true generates spoofed JIDs — DO NOT USE

EDIT MESSAGE
  sock.sendMessage(jid, { text: 'corrected text', edit: originalMsgKey })

DELETE MESSAGE
  sock.sendMessage(jid, { delete: msgKey })

PIN MESSAGE
  sock.sendMessage(jid, { pin: msgKey, type: 1 })  // 1=24h, 2=7d, 3=30d
  sock.sendMessage(jid, { pin: msgKey, type: 0 })  // unpin

KEEP IN CHAT
  sock.sendMessage(jid, { keepInChat: { key: msgKey, keepType: 1 } })

VIEW-ONCE MEDIA
  sock.sendMessage(jid, { image: Buffer, viewOnce: true })
  sock.sendMessage(jid, { video: Buffer, viewOnce: true })

EPHEMERAL (disappearing)
  sock.sendMessage(jid, { text: 'disappears', ephemeralExpiration: 86400 })
  // 86400=24h, 604800=7d, 7776000=90d, 0=off

FORWARD
  sock.relayMessage(jid, message.message, { messageId: key.id })

REPLY (quoted)
  sock.sendMessage(jid, { text: 'reply' }, { quoted: quotedMsg })
